### Inside Walker Agent
```
public void SetWalkingDirection(int directionIndex)
    {
        walkingDirection = (Direction)directionIndex;
        switch (directionIndex)
        {
            case 0:
                SetModel("Walker", modelForward);
                break;
            case 1:
                SetModel("Walker", modelRight);
                break;
            case 2:
                SetModel("Walker", modelBackward);
                break;
            case 3:
                SetModel("Walker", modelLeft);
                break;
        }
    }
```

### Inside User Controll Target
```
void FixedUpdate()
    {
        float inputHor = Input.GetAxis("Horizontal");
        float inputVert = Input.GetAxis("Vertical");
        if (inputVert > 0)
        {
            agent.SetWalkingDirection(0);
        }
        else if (inputVert < 0)
        {
            agent.SetWalkingDirection(2);
        }
        else if (inputHor > 0)
        {
            //normally right 1 but for now change to forward 0
            agent.SetWalkingDirection(0);
        }
        else if (inputHor < 0)
        {
            //normally left 3 but for now change to forward 0
            agent.SetWalkingDirection(3);
        }
        Vector3 hor = transform.right * inputHor;
        hor.y = 0f;
        Vector3 vert = transform.forward * inputVert;
        vert.y = 0f;
        target.transform.position = transform.position + hor + vert;
    }
```
